match_and_combine <- function(data1, col1, data2, col2) {
  d1 <- data1[(data1[,col1] %in% data2[,col2]), ]
  d2 <- data2[(data2[,col2] %in% data1[,col1]), ]
  
  m1 <- match(d1[,col1], d2[,col2])
  
  return(cbind(d1,d2[m1,]))
}

Which are signficantly different between asthmatics and controls? This will have a strong effect on the value of the interpretation. That is, if a new cell type is discovered thats significantly different between asthmatics and controls, thats more valuable in this context than a novel cell type thats the same in each.

An appropriate test for comparing two proportions is a z-test if its parametric and fishers exact test if non-parametric. The boxplots lead me to think non-parametric is most appropriate.

# Create a 2x2 contigency table for each of the cell types in the single cell analysis
z <- 
  df %>%
  group_by(group, cluster.named) %>%
  dplyr::summarize(cells = sum(n),
                   total = sum(batch_total)) %>%
  mutate(not.cells = total - cells)

# 2x2 table for each cluster.named
ct <- list()    
for (c in z$cluster.named) {
   tmp <- z %>%
    filter(cluster.named == c) %>%
    select(group, cells, not.cells) %>%
    data.frame
    
   rownames(tmp) <- tmp$group
   ct[[c]] <- fisher.test(as.matrix(tmp[,-1]))$p.value
}
